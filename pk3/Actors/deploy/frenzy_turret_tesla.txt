			TTRT A 0 A_PlaySoundEx ("TTRTa/Loop", "SoundSlot5", 1)
			
			goto Spawn3

		See:
			NULL A 0 A_JumpIfhealthLower(200,"See2")
			TTRT AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 1 A_Chase
			TTRT A 5  A_PlaySound("turret/active", CHAN_AUTO, 6)
			goto See
		See2:
			NULL A 0 A_SpawnItemEx("Smoke",random(-8,8),random(-8,8),56,0,0,random(1,3))
			TTRT AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 1 A_Chase
			TTRT A 5  A_PlaySound("turret/active", CHAN_AUTO, 6)
			goto See2
		Missile:
			NULL A 0 A_JumpIfHealthLower(300,"Missile2")
			TTRT A 2 A_FaceTarget
			//TTRT A 0 bright A_PlaySound ("TTRTa/Attack")
			TTRT A 0 bright A_PlaySound("x1/taser") //<---------------------------- Dropped here sound.. its stupid when its silent
			TTRT A 0 bright A_SpawnItem ("TTRTaZapper", 0, 60)
			TTRT AAAA 1 bright
			TTRT A 2
			goto See
		Missile2:
			TTRT A 2 A_FaceTarget
			//TTRT A 0 bright A_PlaySound ("TTRTa/Attack")
			TTRT A 0 bright A_PlaySound("x1/taser") //<---------------------------- Dropped here sound.. its stupid when its silent

			NULL A 0 A_SpawnItemEx("Smoke",random(-8,8),random(-8,8),56,0,0,random(1,3))
			TTRT AAAA 1 bright
			TTRT A 2
			goto See
		
		Pain:
			TTRT A 1
			TTRT A 1 A_Pain
			goto See
		Death:
			//BTRT A 1 bright A_SpawnItemEx ("Boom", 0, 0, 60, 0, 0, 0, 0, 160)
			TTRT A 0
			TTRT A 0 A_KillChildren
                        NULL A 0 A_NoBlocking
			TTRT A 1 bright A_SpawnItemEx("explosionemitter_t", 0)
			TTRT A 0 bright A_StopSoundEx ("SoundSlot5")
			TTRT A 0 bright A_PlaySound ("weapons/rocklx")
			TTRT B 1
			
                       		NULL A 0 A_NoBlocking
			NULL A 0 A_KillMaster
			POSS A 0 A_SpawnDebris("splatSandbag1")
			POSS A 0 A_SpawnDebris("splatSandbag2")
			POSS A 0 A_SpawnDebris("splatSandbag3")
			POSS A 0 A_SpawnDebris("splatSandbag1")
			POSS A 0 A_SpawnDebris("splatSandbag2")
			POSS A 0 A_SpawnDebris("splatShard1")
			POSS A 0 A_SpawnDebris("splatShard2")
			POSS A 0 A_SpawnDebris("splatShard3")
			POSS A 0 A_SpawnDebris("splatShard4")
