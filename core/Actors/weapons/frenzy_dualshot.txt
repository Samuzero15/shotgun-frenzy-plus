ACTOR Dualshot : Weapon
{
	Weapon.AmmoUse 			1
	Weapon.AmmoGive 		50
	Weapon.AmmoType			"Dualshot_Ammo"
	Inventory.PickupMessage		"You got the Dualshot!"
	
	+WEAPON.NOAUTOFIRE
	
	Weapon.UpSound			"w7/load"
	weapon.selectionorder 2900
	
	States
	{
		Ready:
			X4E1 A 5 A_WeaponReady
			goto Ready
		Deselect:
			X4E1 A 1 A_Lower
			X4E1 A 0 A_Lower
			goto Deselect
		Select:
			X4E1 A 1 A_Raise
			X4E1 A 0 A_Raise
			goto Select
		Fire:
			X4E1 B 2
			X4E1 C 2 A_PlaySound("w2/fire2", CHAN_WEAPON)
			X4E1 D 0 A_FireCustomMissile("Dualshot_Nail", 0, 1, -8, 0, 0)
			X4E1 D 0 A_FireCustomMissile("Dualshot_Nail", 0, 1, 8, 0, 0)
			X4E1 E 2
			X4E1 A 2
			X4E1 A 0 A_Refire
    			goto Ready
		
		FireRockets:
			TNT1 A 1
		goto lightdone
	}
}
