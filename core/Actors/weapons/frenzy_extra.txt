ACTOR BoomerShot : fastprojectile
{
  Game Doom
  Radius 32
  Height 32
  Speed 125
  Damage 65
  RenderStyle Add
  Alpha 0.75
  scale 2.0
  SeeSound "x3/fire1"
  DeathSound "baby/shotx"
  Projectile
  +RANDOMIZE
  states
  {
  Spawn:
    APLS AB 5 bright
    loop
  Death:
    APBX ABCDE 2 bright
    stop
  }
}


ACTOR BoomerCannon : Weapon
{
	//Weapon.AmmoUse			1
	//Weapon.AmmoGive 		50
	//Weapon.AmmoType			"Cell"
	Inventory.PickupMessage		"You got the Flamer 2.0!"
	Obituary			"%o got flamed by a fanboy"
	
	Weapon.UpSound			"w/load6"
	AttackSound			"x3/fire1"
	
	States
	{
		Ready:
			X3E1 A 1 A_WeaponReady(WRF_DISABLESWITCH)
			loop
		Deselect:
			X3E1 A 1 A_Lower
			loop
		Select:
			X3E1 A 1 A_Raise
			loop
		Fire:
			X3E1 B 1 bright A_FireCustomMissile("BoomerShot", 0)
			X3E1 C 2 bright
			X3E1 A 1 A_Refire
			goto Ready
		AltFire:
			X3E1 A 1
			X3E1 A 1 ACS_Execute(95, 0, 0, 0, 0)
			goto Ready
		Flash:
			X3E1 C 4 Bright A_Light1
			goto LightDone
			X3E1 C 4 Bright A_Light1
			goto LightDone
			
		FireRockets:
			TNT1 A 1
		goto lightdone
	}
}
