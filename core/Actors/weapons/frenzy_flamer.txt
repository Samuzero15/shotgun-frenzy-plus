ACTOR Flamer : Weapon
{
	Weapon.AmmoUse			1
	Weapon.AmmoGive 		50
	Weapon.AmmoType			"Gasoline"
	Inventory.PickupMessage		"You got the Flamer 2.0!"
	Obituary			"%o got flamed by a fanboy"
	
	Weapon.UpSound			"w/load6"
	AttackSound			"x2/fire1"
	weapon.selectionorder 2800
	States
	{
		Ready:
			BLST A 1 A_WeaponReady
			goto Ready
		Deselect:
			BLST A 1 A_Lower
			BLST A 0 A_Lower
			goto Deselect
		Select:
			BLST A 1 A_Raise
			BLST A 0 A_Raise
			goto Select
		Fire:
			BLST B 5 A_FireCustomMissile("FlamerShot", 0)
			TNT1 A 0 A_FireCustomMissile("FlamerShot", 5)
			TNT1 A 0 A_FireCustomMissile("FlamerShot", -5)
			BLST B 4 
			BLST C 20 A_Refire
			goto Ready
		Flash:
			BLST C 4 Bright A_Light1
			goto LightDone
			BLST C 4 Bright A_Light1
			goto LightDone
			
		FireRockets:
			TNT1 A 1
		goto lightdone
	}
}
