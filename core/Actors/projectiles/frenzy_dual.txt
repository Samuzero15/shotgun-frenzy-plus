Actor Dualshot_Nail
{
        Projectile
        Radius 6
        Height 8
        Speed 30
        Damage (10)
        Scale 0.6
        +BLOODSPLATTER
		+STRIFEDAMAGE
		+MTHRUSPECIES
		+ripper
        seeSound ""
        DeathSound ""
        Decal Bulletchip
        DamageType MarineFire2
		
		var int user_riptime;
		
        States
        {
        Spawn:
			TNT1 A 0 A_JumpIf(user_riptime >= 35, "StopRippin")
            DART A 1 A_SpawnItemEx("NailBlur", 0, 0, 0, 0.2, 0, 0, 180, 128)
        	DART A 0 A_PlaySound("weapons/nailflight")
			DART A 1 A_SpawnItemEx("NailBlur", 0, 0, 0, 0.2, 0, 0, 180, 128)
			TNT1 A 0 A_SetUserVar("user_riptime", user_riptime+1)
            goto Spawn
		StopRippin:
			TNT1 A 0 A_ChangeFlag("Ripper", false)
			DART A 1 A_SpawnItemEx("NailBlur", 0, 0, 0, 0.2, 0, 0, 180, 128)
        	DART A 0 A_PlaySound("weapons/nailflight")
			DART A 1 A_SpawnItemEx("NailBlur", 0, 0, 0, 0.2, 0, 0, 180, 128)
			goto StopRippin+1
        Crash:
        	DART B 0 A_PlaySound("weapons/nailhit")
        	DART BCDEFG 2
        	Stop
        XDeath:
        	DART B 0 A_PlaySound("weapons/nailhitbleed")
        	DART BCDEFG 2
        	Stop
        Death:
        	DART B 0 A_PlaySound("weapons/nailhit")
        	DART BCDEFG 2
        	Stop
        }
}
