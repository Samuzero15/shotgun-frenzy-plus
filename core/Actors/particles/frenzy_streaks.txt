// particle actors
actor streak_red
{
	renderstyle add
	
	+FORCEYBILLBOARD
	+CLIENTSIDEONLY
	+NOINTERACTION
	
	States
	{
		Spawn:
			TNT1 A 1 NoDelay ACS_NamedExecuteAlways("ParticleCheck", 0, 2)
			TNT1 A 1
			TNT1 A 50 A_Jump(256, "Loopme")
		Loopme:
			STRK A 1 bright A_FadeOut(0.02)
			loop
		NoSpawn:
			TNT1 A 1
			Stop
	}
}

ACTOR streakQuickEmitter_Yellow 
{
	renderstyle none
	
	+NOCLIP
	+NOGRAVITY
	+ClientSideOnly
	var int user_parts;
	
	States
	{
		Spawn:
			TNT1 A 1
			TNT1 A 1 A_SetUserVar("user_parts", 10)
		Loopme:
			TNT1 A 0 A_SpawnItemEx("streak_yellow", random(-32, 32), random(-32, 32), 0, 0, 0, 3, 0, SXF_CLIENTSIDE, 0)
			TNT1 A 0 A_Warp(AAPTR_Target, 0,0,0,0,WARPF_NOCHECKPOSITION|WARPF_INTERPOLATE)
			TNT1 A 2 A_SetUserVar("user_parts", user_parts-1)
			TNT1 A 1 A_JumpIf(user_parts <= 0, "Death")
			goto Loopme
		Death:
			TNT1 A 1
			stop
	}
}
